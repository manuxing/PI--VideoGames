describe("Solution Tests", function() {

  let dinglemouseAnswer20170821_antBridge = function(ants, terrain) {
  
    let isGap = function(x, ary) {
      return ary[x] == '.' || (x > 0 && ary[x-1] == '.') || (x < ary.length-1 && ary[x+1] == '.');
    };
    
    let leftmostAnt = function(ary) {
      let i = 0;
      for (let c of ary) if (c == ' ') i++; else break;
      return i;
    };
    
    // This is a trick. Reverse the initial string, so then can ALWAYS move the ants from the back!
    let reverseAnts = ants.split('').reverse().join('');
    
    let antlen = ants.length;
    let landlen = terrain.length;
    
    let antChars = [];
    for (let i = 0; i < antlen+landlen; i++) antChars.push(' ');
    for (let i = 0; i < reverseAnts.length; i++) antChars[i] = reverseAnts[i];
    
    let landChars = [];
    for (let i = 0; i < antlen; i++) landChars.push('~');
    for (let i = 0; i < landlen; i++) landChars.push(terrain[i]);
    
    let result = "";
    while (result.length != ants.length) {
      // Get the left-most ant and move him forward as far as he can go
      // If he can get all the way to the end, add him to the result
      let leftAnt = leftmostAnt(antChars);
      for (let i = leftAnt; i < landChars.length; i++) {
          
        // part of a bridge
        if (isGap(i, landChars) && antChars[i] == ' ') {
          antChars[i] = antChars[leftAnt];
          antChars[leftAnt] = ' ';
          break;
        }        
  
        // made it to the other side
        if (i == landChars.length - 1) {
          result = antChars[leftAnt] + result;
          antChars[leftAnt] = ' ';   
          break;
        }
        
      } // for      
    } // while
    
    return result;
  };
    
  let makeRandomAnts = function() {
    let ants = "ABCDEFGHIJKLMNOPQRSTUVWXYZ".substring(0, 3 + Math.floor(Math.random() * 10));
    return ants.split('').reverse().join('');
  };
  
  const SOLID = "------------------------------";
  const GAP = "..............................";
  let makeRandomTerrain = function(nants) {
    let s = SOLID.substring(0, 1 + Math.floor(Math.random() * 6));
    let gaps = Math.floor(Math.random() * 6); // 0-5 gaps
    while (gaps-- > 0) {
      s += GAP.substring(0, 1 + Math.floor(Math.random() * (nants - 2)));
      s += SOLID.substring(0, 1 + Math.floor(Math.random() * 6));
    } 
    return s;
  };

    // With visualization (in node js, delete some comments below)
function antBridge(ants, terrain) {
  const terra = " ".repeat(ants.length) + terrain + " ".repeat(ants.length);
  let bridges = "";
  for (let i = 0; i < terra.length; i++) {
    if (terra[i] === "." || terra[i + 1] === "." || terra[i - 1] === ".") {
      bridges += "^";
    } else bridges += " ";
  }
  let antsArr = [];
  for (let i = 0; i < ants.length; i++) {
    antsArr.push([ants[i], i]);
  }
  let antsToShow = newAntsToShow(antsArr, terra);
  let pos = 0;
  while (antsToShow.search(RegExp(`[A-Z]{${antsArr.length}}$`)) === -1) {
    for (let i = 0; i < ants.length; i++) {
      // MOVES:
      // not on the bridge and nobody in front of us
      if (
        bridges[antsArr[i][1]] === " " &&
        antsToShow[antsArr[i][1] + 1] === " "
      ) {
        antsArr[i][1]++;
        antsToShow = newAntsToShow(antsArr, terra, bridges);
        break;
      }
      // not on the bridge, bridge with ant in front of us;
      // if bridge and place right after it is taken - skip,
      // otherwise - move
      if (
        bridges[antsArr[i][1]] === " " &&
        bridges[antsArr[i][1] + 1] !== " " &&
        antsToShow[antsArr[i][1] + 1] !== " "
      ) {
        pos = antsArr[i][1] + 1;
        while (antsToShow[pos] !== " " && bridges[pos] !== " ") {
          pos++;
        }
        if (bridges[pos - 1] !== " " && antsToShow[pos] === " ") {
          antsArr[i][1] = pos;
          antsToShow = newAntsToShow(antsArr, terra, bridges);
          break;
        }
      }
      // on the bridge, nobody behind us,
      // the bridge upfront or place right after it is free
      if (bridges[antsArr[i][1]] !== " " && i === 0) {
        pos = antsArr[i][1] + 1;
        while (antsToShow[pos] !== " " && bridges[pos] !== " ") {
          pos++;
        }
        if (antsToShow[pos] === " ") {
          antsArr[i][1] = pos;
          antsToShow = newAntsToShow(antsArr, terra, bridges);
          break;
        }
      }
    }
    // array sorting after every move
    antsArr.sort((a, b) => a[1] - b[1]);
  }
  return antsToShow.trim();
}

function newAntsToShow(arr, terra, bridge) {
  let antsPattern = " ".repeat(terra.length);
  for (let i = 0; i < arr.length; i++) {
    antsPattern =
      antsPattern.slice(0, arr[i][1]) +
      arr[i][0] +
      antsPattern.slice(arr[i][1] + 1);
  }
  // *** delay (in ms) and visualization ***
  // *** delete comments below to visualize
  // sleep(30);
  // console.log("\033[2J");
  // console.log(antsPattern);
  // console.log(terra);
  // console.log(bridge);
  return antsPattern;
}

function sleep(milliseconds) {
  const date = Date.now();
  let currentDate = null;
  do {
    currentDate = Date.now();
  } while (currentDate - date < milliseconds);
}
